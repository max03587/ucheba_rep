    def __init__(self, n = None):
        self.prev = None
        self.next = None
        self.name = n
        
    def __str__(self):
        return self.name

class cat_LinkedList:

    def __init__(self):
        self.head = None
        self.tail = None

    def add(self, cat_name):
        newCat = cat_Node(cat_name)

        if self.head  is None:
            self.head = newCat
            
        else:
            current = self.head
            
            while current.next is not None:
                current = current.next
            current.next = newCat

  

CATS = cat_LinkedList()
